📋 CSS for Hero Morph (both pages)

/* === Pattern 2: Hero Morph === */
/* Page 1 (source) */

@view-transition {
  navigation: auto;
}

.hero { view-transition-name: hero-image; }
/* Element: 200×200px square, border-radius 12px */

/* --- */

/* Page 2 (destination) */

@view-transition {
  navigation: auto;
}

.hero { view-transition-name: hero-image; }
/* Element: 120×120px circle, border-radius 50% */

/* Browser automatically animates the size,
   shape, and position difference between
   the two named elements. No keyframes needed. */
