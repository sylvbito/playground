/* === Pattern 1: Default Cross-Fade === */
/* Add to both source and destination pages */

@view-transition {
  navigation: auto;
}

/* Optional: adjust duration */
::view-transition-old(root),
::view-transition-new(root) {
  animation-duration: 0.4s;
}

/* That's it. The browser cross-fades between
   snapshots of the old and new pages. No JS. */
